# MySql操作

# 系统操作

# 创建用户

只允许指定ip连接

create user '新用户名'@'localhost' identified by '密码';

//允许所有ip连接（用通配符%表示）

create user '新用户名'@'%' identified by '密码';

# 授权，撤权

GRANT 权限 ON databasename.tablename TO 'username'@'host';

REVOKE 权限 ON databasename.tablename FROM 'username'@'host';

全部权限就是ALL，全部库和表就是*.*

# 更改用户密码

SET PASSWORD FOR 'username'@'host' = PASSWORD('newpassword');

//修改当前用户密码SET PASSWORD = PASSWORD("newpassword");

# 删除用户

DROP USER ‘username’@'host’;

# 修改用户连接ip

rant all privileges on . to root@"%" identified by "密码";

flush privileges;  刷新权限

(+) : 这个不能和join一起用,只能出现在where子句中; 可以用在笛卡尔积的写法上 , 哪个表后面有(+)哪个作为主表
