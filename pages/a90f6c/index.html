# Docker

# docker安装

1.卸载之前的docker

yum remove docker \
           docker-client \
           docker-client-latest \
           docker-common \
           docker-latest \
           docker-latest-logrotate \
           docker-logrotate \
           docker-engine

2.下载依赖包

yum install -y yum-utils

3.设置下载地址

yum-config-manager \
    --add-repo \
    https://download.docker.com/linux/centos/docker-ce.repo

4.安装docker

yum install docker-ce docker-ce-cli containerd.io

5.启动docker

systemctl start docker

6.设置开机自启

systemctl enable docker

7.配置镜像加速

mkdir -p /etc/docker

tee /etc/docker/daemon.json <<-'EOF'
{
  "registry-mirrors": ["https://12wwwyap.mirror.aliyuncs.com"]
}
EOF

systemctl daemon-reload

systemctl restart docker

# docker常用命令

# 获取镜像

//搜索镜像
docker search ubuntu
//从仓库拉取镜像,如果不指定版本默认下载latest版本
docker pull ubuntu

# 启动容器

docker run -it ubuntu /bin/bash
-i: 交互式操作。
-t: 终端。
-d：后台运行
ubuntu: ubuntu 镜像。
/bin/bash：放在镜像名后的是命令，这里我们希望有个交互式 Shell，因此用的是 /bin/bash

# 进入容器

docker attach id   退出终端会导致容器停止
docker exec -it 710 /bin/bash  进入该id容器，退出终端不会导致容器停止

# 管理容器

docker ps 查看进程

docker ps -a 查看所有进程包括已停止的

docker start CONTAINER ID 重启这个id的容器

docker stop 停止某个容器

docker rm -f redis 删除名为“redis”的容器, -f 可以强行删除正在运行的容器

docker container prune 删除所有停止的容器

docker port id 查看该id容器的所有端口

docker top id 可以查看容器的进程

docker images 查看下载的镜像

docker rmi redis 删除镜像

docker logs -f elasticsearch 查看elasticsearch容器的日志, -f 循环输出

docker update redis –restart=always 容器自动启动

docker run -p 6379:6379 –name redis -v /mydata/redis/data:/data -v /mydata/redis/conf/redis.conf:/etc/redis/redis.conf -d redis redis-server /etc/redis/redis.conf –requirepass “redis94264” 启动redis,并挂在相关的配置文件, 并设置密码redis94264

docker run -p 3306:3306 –name mysql -v /mydata/mysql/log:/var/log/mysql -v /mydata/mysql/data:/var/lib/mysql -v /mydata/mysql/conf:/etc/mysql -e MYSQL_ROOT_PASSWORD=root -d mysql:5.7 启动mysql

# 导出容器

docker export 1e560fca3906 > ubuntu.tar   把该容器快照导出到本地文件

# 导入容器

可以将容器快照文件再导入成镜像

cat docker/ubuntu.tar | docker import - test/ubuntu:v1     从本地导入
docker import http://example.com/exampleimage.tgz example/imagerepo    通过url导入

# 保存镜像

docker save 1e560fca3906 > ubuntu.tar
//组合保存

# 载入镜像

docker load < ubuntu.tar

# 容器互联

创建一个新的docker网络

[root@VM-0-9-centos ~]## docker network create -d bridge test-net

展示所有的docker网络

[root@VM-0-9-centos ~]## docker network ls
NETWORK ID          NAME                DRIVER              SCOPE
5c7a4fccca19        bridge              bridge              local
b49022ce4237        host                host                local
60f9070b901a        none                null                local
ccd7f95f8b63        test-net            bridge              local

创建容器时指明 –network

[root@VM-0-9-centos ~]## docker run -itd --name net1 --network test-net net/ubuntu:v1 /bin/bash
[root@VM-0-9-centos ~]## docker run -itd --name net2 --network test-net net/ubuntu:v1 /bin/bash

这样同一个docker网络内的容器可以互联

# 镜像更新

首先需要启动一个容器,然后进入容器内

执行 apt-get update 更新容器, 退出容器,可以提交容器副本,commit后便会有一个新的镜像

[root@VM-0-9-centos ~]## docker commit -m="test update" -a="test" ae735c1134de test/ubuntu:v2
sha256:b52d4b375f0e765c8ee18dee0c2be0e6d49fd0b60d8b179cf42204071b90f244

-m: 提交的描述信息

-a: 指定镜像作者

e218edb10161：容器 ID

runoob/ubuntu:v2: 指定要创建的目标镜像名

# 镜像创建

创建一个 Dockerfile 文件，其中包含一组指令来告诉 Docker 如何构建我们的镜像

[root@VM-0-9-centos ~]## vim Dockerfile
runoob@runoob:~$ cat Dockerfile
FROM    centos:6.7
MAINTAINER      Fisher "fisher@sudops.com"

RUN     /bin/echo 'root:123456' |chpasswd
RUN     useradd runoob
RUN     /bin/echo 'runoob:123456' |chpasswd
RUN     /bin/echo -e "LANG=\"en_US.UTF-8\"" >/etc/default/local
EXPOSE  22
EXPOSE  80
CMD     /usr/sbin/sshd -D

第一条FROM，指定使用哪个镜像源

RUN 指令告诉docker 在镜像内执行命令，安装了什么。。。

docker build 进行构建

[root@VM-0-9-centos ~]## docker build -t zsk/centos:6.7 .

-t ：指定要创建的目标镜像名

. ：Dockerfile 文件所在目录，可以指定Dockerfile 的绝对路径

# 更新镜像tag

会复制一个镜像,不是覆盖原有

[root@VM-0-9-centos ~]## docker tag 633c4e471363 zsk/centos:dev

# 安装实例

# 安装ES和kibana

# 下载

docker pull elasticsearch:7.4.2
docker pull kibana:7.4.2

# 安装

# ES

在主机创建外挂目录:

mkdir -p /mydata/elasticsearch/config
mkdir -p /mydata/elasticsearch/data
echo "http.host: 0.0.0.0">>/mydata/elasticsearch/config/elasticsearch.yml
chmod -R 777 /mydata/elasticsearch  //修改文件权限

docker run --name elasticsearch -p 9200:9200 -p 9300:9300 \
-e "discovery.type=single-node" \
-e ES_JAVA_OPTS="-Xms64m -Xmx256m" \
-v /mydata/elasticsearch/config/elasticsearch.yml:/usr/share/elasticsearch/config/elasticsearch.yml \
-v /mydata/elasticsearch/data:/usr/share/elasticsearch/data \
-v /mydata/elasticsearch/plugins:/usr/share/elasticsearch/plugins \
-d elasticsearch:7.4.2

# kibana

docker run --name kibana -e ELASTICSEARCH_HOSTS=http://119.45.176.143:9200 -p 5601:5601 \
-d kibana:7.4.2

# 安装nginx

现在mydata下准备一套conf (安装一套nginx骗取他的文件在清理掉, docker container cp nginx:/etc/nginx .)

docker run -p 80:80 --name bginx \
-v /mydata/nginx/html:/usr/share/nginx/html \
-v /mydata/nginx/logs:/var/log/nginx \
-v /mydata/nginx/conf:/etc/nginx \
-d nginx:1.10

# 安装RocketMQ

启动NameServer：

docker run -d -p 9876:9876 --name rmqserver  foxiswho/rocketmq:server-4.5.1

启动broker：

docker run -d -p 10911:10911 -p 10909:10909\
 --name rmqbroker --link rmqserver:namesrv\
 -e "NAMESRV_ADDR=namesrv:9876" -e "JAVA_OPTS=-Duser.home=/opt"\
 -e "JAVA_OPT_EXT=-server -Xms128m -Xmx128m"\
 foxiswho/rocketmq:broker-4.5.1

Broker容器中默认的配置文件的路径为：

安装后需要进入该文件添加 brokerIP1 = 192.168.53.201 不然默认是内网IP

/etc/rocketmq/broker.conf

也可以通过-v参数指定本机的配置文件：

docker run -d -p 10911:10911 -p 10909:10909\ --name rmqbroker --link rmqserver:namesrv\ -e "NAMESRV_ADDR=namesrv:9876" -e "JAVA_OPTS=-Duser.home=/opt"\ -e "JAVA_OPT_EXT=-server -Xms128m -Xmx128m"\ -v /conf/broker.conf:/etc/rocketmq/broker.conf \ foxiswho/rocketmq:broker-4.5.1

如果一切正常，NameServer和Broker一会儿就会安装好，为了管理上的方便，rocketmq console也是必不可少的工具了，通过上面查询的方式找到需要启动的版本，启动方式如下：

管理页面访问端口是 8180 ,可修改配置

docker run -d --name rmqconsole -p 8180:8080 --link rmqserver:namesrv\ -e "JAVA_OPTS=-Drocketmq.namesrv.addr=namesrv:9876\ -Dcom.rocketmq.sendMessageWithVIPChannel=false"\ -t styletang/rocketmq-console-ng

然后通过如下命令检查一下启动情况

docker ps|grep rocketmq
